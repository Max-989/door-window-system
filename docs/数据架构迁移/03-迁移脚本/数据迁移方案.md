### 3.3 枚举转换

旧系统的单选字段（如 `singleSelect`）需按以下规则转换：
- 将中文选项值映射为新系统的英文 enum 值
- 无法映射的值记录到 `migration_errors` 表供人工审核

### 3.4 默认值

| 场景 | 默认值 |
|---|---|
| 空状态 | pending |
| 空金额 | 0 |
| 空数量 | 0 |
| 空备注 | '' |
| FK 找不到关联 | null + 记录警告 |
| 时间戳为空 | 当前迁移时间 |

### 3.5 图片/附件处理

- .base 中的 attachment 类型字段存储为飞书文件 token
- 迁移时先下载原文件，上传到新系统存储，记录新 URL
- 下载失败的记录到错误日志，标记为待处理

### 4.2 .base 文件解析器

```python
# base_parser.py
"""柯尚订单管理.base 文件解析器"""
import gzip
import json
import base64
from pathlib import Path
from typing import Dict, List, Any


class BaseFileParser:
    """飞书多维表格 .base 文件解析器"""

    def __init__(self, file_path: str):
        self.file_path = Path(file_path)
        self.data: Dict[str, Any] = {}
        self.tables: Dict[str, Any] = {}

    def parse(self) -> Dict[str, Any]:
        """解析 .base 文件，返回 {表名: [行数据]}"""
        raw = self.file_path.read_bytes()

        # 解析 JSON
        doc = json.loads(raw)
        snapshot = doc.get('gzipSnapshot', '')
        if not snapshot:
            raise ValueError('.base 文件中未找到 gzipSnapshot 字段')

        # Base64 解码 + GZIP 解压
        compressed = base64.b64decode(snapshot)
        decompressed = gzip.decompress(compressed)
        self.data = json.loads(decompressed)

        # 提取表结构
        self._extract_tables()
        return self.tables

    def _extract_tables(self):
        """从 schema.tableMap 中提取表定义和数据"""
        for element in self.data:
            if not isinstance(element, dict):
                continue

            schema = element.get('schema', {})
            table_map = schema.get('tableMap', {})

            for table_id, table_info in table_map.items():
                table_name = table_info.get('name', f'unknown_{table_id}')
                fields = table_info.get('fieldMap', {})

                # 提取字段定义
                field_defs = {}
                for fid, finfo in fields.items():
                    field_defs[finfo.get('name', fid)] = {
                        'id': fid,
                        'type': finfo.get('type', 'text'),
                    }

                self.tables[table_name] = {
                    'id': table_id,
                    'fields': field_defs,
                    'data': [],  # 数据行将在后续填充
                }

            # 提取数据行（recordMap）
            record_map = element.get('recordMap', {})
            if record_map:
                # 将记录分配到对应的表
                pass  # 具体分配逻辑需根据实际数据结构补充

    def get_table_names(self) -> List[str]:
        return list(self.tables.keys())

    def get_table(self, table_name: str) -> Dict:
        return self.tables.get(table_name, {})
```

### 4.3 主迁移命令

```python
# migrate_data.py
"""
Django Management Command: 数据迁移

使用方式:
    python manage.py migrate_data --base-file /path/to/柯尚订单管理.base --batch-size 100 --dry-run
    python manage.py migrate_data --base-file /path/to/柯尚订单管理.base --tables brand,order --phase 1
    python manage.py migrate_data --rollback --phase 1
"""
import gzip
import json
import base64
import logging
from django.core.management.base import BaseCommand
from django.db import transaction
from typing import Optional

logger = logging.getLogger(__name__)

# 迁移阶段定义（按依赖顺序）
MIGRATION_PHASES = {
    1: ['装企目录', '产品信息', '配件目录'],           # 基础数据（无 FK 依赖）
    2: ['安装师傅表', '销售订单'],                     # 人员 + 订单
    3: ['订单明细', '发货记录'],                        # 订单关联
    4: ['安装单', '维修表', '量尺任务'],                # 流程数据
    5: ['付款记录表', '五金进销存', '客户跟进记录表'],    # 交易与辅助
}

# 表名 → Transformer 映射
TABLE_TRANSFORMERS = {
    '装企目录': 'transformers.brand_transformer.migrate_brands',
    '产品信息': 'transformers.product_transformer.migrate_products',
    '安装师傅表': 'transformers.worker_transformer.migrate_workers',
    '销售订单': 'transformers.order_transformer.migrate_orders',
    '安装单': 'transformers.installation_transformer.migrate_installations',
    '维修表': 'transformers.maintenance_transformer.migrate_maintenance',
}

# 旧表 → 新模型映射（用于 FK 查找）
OLD_TO_NEW_ID_MAP = {}  # 运行时填充: {(old_table, old_id): new_model_instance}


class Command(BaseCommand):
    help = '从 .base 文件迁移数据到新数据库'

    def add_arguments(self, parser):
        parser.add_argument('--base-file', type=str, required=True,
                          help='.base 文件路径')
        parser.add_argument('--batch-size', type=int, default=100,
                          help='每批处理数量（默认 100）')
        parser.add_argument('--tables', type=str, default='',
                          help='指定迁移表名（逗号分隔），为空则全部')
        parser.add_argument('--phase', type=int, default=0,
                          help='指定迁移阶段（1-5），为空则全部')
        parser.add_argument('--dry-run', action='store_true',
                          help='试运行，不实际写入数据库')
        parser.add_argument('--rollback', action='store_true',
                          help='回滚指定阶段数据')
        parser.add_argument('--skip-errors', action='store_true',
                          help='跳过错误记录继续执行')

    def handle(self, *args, **options):
        base_file = options['base_file']
        batch_size = options['batch_size']
        dry_run = options['dry_run']
        rollback = options['rollback']

        self.stdout.write(f'{"=" * 60}')
        self.stdout.write(f'门窗系统数据迁移')
        self.stdout.write(f'{"=" * 60}')
        self.stdout.write(f'文件: {base_file}')
        self.stdout.write(f'模式: {"DRY-RUN" if dry_run else "LIVE"}')
        self.stdout.write(f'批大小: {batch_size}')

        # 1. 解析 .base 文件
        self.stdout.write('\n[1/4] 解析 .base 文件...')
        parser = BaseFileParser(base_file)
        try:
            tables = parser.parse()
            self.stdout.write(self.style.SUCCESS(
                f'  成功解析 {len(tables)} 个表'))
            for name in tables:
                self.stdout.write(f'    - {name}')
        except Exception as e:
            self.stdout.write(self.style.ERROR(f'  解析失败: {e}'))
            return

        # 2. 确定迁移范围
        tables_to_migrate = self._get_tables_to_migrate(
            options, tables.keys())
        self.stdout.write(f'\n[2/4] 迁移范围: {len(tables_to_migrate)} 个表')

        # 3. 执行迁移或回滚
        if rollback:
            self._do_rollback(options['phase'], dry_run)
        else:
            self._do_migrate(tables_to_migrate, tables, batch_size, dry_run,
                           options['skip_errors'])

        # 4. 输出统计
        self._print_summary()

    def _get_tables_to_migrate(self, options, available_tables):
        """确定要迁移的表"""
        phase = options.get('phase', 0)
        tables_str = options.get('tables', '')

        if tables_str:
            specified = [t.strip() for t in tables_str.split(',')]
            return [t for t in specified if t in available_tables]

        if phase:
            phases_to_run = {p: t for p, t in MIGRATION_PHASES.items()
                           if p >= phase}
        else:
            phases_to_run = MIGRATION_PHASES

        result = []
        for p in sorted(phases_to_run):
            for t in phases_to_run[p]:
                if t in available_tables:
                    result.append(t)
        return result

    def _do_migrate(self, tables, parsed_data, batch_size, dry_run,
                   skip_errors):
        """执行迁移"""
        global OLD_TO_NEW_ID_MAP

        for table_name in tables:
            self.stdout.write(f'\n  迁移表: {table_name}')

            transformer_path = TABLE_TRANSFORMERS.get(table_name)
            if not transformer_path:
                self.stdout.write(self.style.WARNING(
                    f'    跳过：无对应 transformer'))
                continue

            data = parsed_data[table_name]
            rows = data.get('data', [])
            total = len(rows)

            if total == 0:
                self.stdout.write(f'    无数据，跳过')
                continue

            self.stdout.write(f'    共 {total} 条记录')

            # 分批处理
            success = 0
            errors = 0
            for i in range(0, total, batch_size):
                batch = rows[i:i + batch_size]
                try:
                    if not dry_run:
                        with transaction.atomic():
                            self._import_batch(
                                transformer_path, batch, table_name)
                    success += len(batch)
                except Exception as e:
                    if skip_errors:
                        errors += len(batch)
                        logger.error(f'{table_name} 批次 {i}: {e}')
                    else:
                        raise

                progress = min(i + batch_size, total)
                self.stdout.write(
                    f'    进度: {progress}/{total} '
                    f'(成功 {success}, 失败 {errors})')

            self.stdout.write(self.style.SUCCESS(
                f'    完成: 成功 {success}, 失败 {errors}'))

    def _import_batch(self, transformer_path, batch, table_name):
        """导入一批数据"""
        module_path, func_name = transformer_path.rsplit('.', 1)
        module = __import__(module_path, fromlist=[func_name])
        func = getattr(module, func_name)
        func(batch, table_name)

    def _do_rollback(self, phase, dry_run):
        """回滚数据"""
        # 回滚逻辑见第 6 节
        pass

    def _print_summary(self):
        """输出迁移统计"""
        pass


# 需要在顶部导入
from migration_config import BaseFileParser
```

```python
"""装企 → Brand + Store + DecorationStaff"""
from decoration.models import Brand, Store, DecorationStaff
from migration_config import OLD_TO_NEW_ID_MAP


def migrate_brands(rows, table_name):
    """迁移装企数据"""
    for row in rows:
        fields = row.get('fields', {})

        # 创建 Brand
        brand = Brand.objects.create(
            name=fields.get('装企名称', ''),
            contact_person=fields.get('联系人', '') or None,
            phone=fields.get('联系电话', '') or None,
            status=_map_status(fields.get('状态', '启用')),
        )

        # 记录 ID 映射
        OLD_TO_NEW_ID_MAP[(table_name, row.get('recordId'))] = brand

        # 如果有门店信息，创建 Store
        store_name = fields.get('门店名称', '')
        if store_name:
            store = Store.objects.create(
                brand=brand,
                name=store_name,
                address=fields.get('门店地址', fields.get('地址', '')),
                city=fields.get('城市', ''),
            )
            OLD_TO_NEW_ID_MAP[(table_name + '_store', row.get('recordId'))] = store


def _map_status(old_value):
    status_map = {
        '启用': 'active',
        '合作中': 'active',
        '已暂停': 'paused',
        '停用': 'inactive',
        '已终止': 'inactive',
    }
    return status_map.get(old_value, 'active')
```

```python
def migrate_order_items(rows, table_name):
```

## 5. 数据校验方案

### 5.1 迁移前校验

```python
# validators/data_validator.py
"""数据校验器"""

class MigrationValidator:
    """迁移前后数据一致性检查"""

    def validate_counts(self, old_tables, models):
        """数量校验：旧表行数 vs 新模型记录数"""
        for table_name, model in models.items():
            old_count = len(old_tables.get(table_name, {}).get('data', []))
            new_count = model.objects.count()
            status = '✅' if old_count == new_count else '❌'
            print(f'{status} {table_name}: 旧={old_count} 新={new_count}')

    def validate_data_integrity(self, model, checks):
        """数据完整性校验"""
        for field, validator in checks.items():
            null_count = model.objects.filter(**{field: None}).count()
            empty_count = model.objects.filter(**{field: ''}).count()
            print(f'  {field}: null={null_count}, empty={empty_count}')

    def validate_foreign_keys(self, model, fk_fields):
        """外键完整性校验"""
        for field in fk_fields:
            # 检查指向不存在的 FK
            broken = model.objects.filter(
                **{f'{field}__isnull': False}
            ).exclude(
                **{f'{field}__isnull': False}  # 确保关联对象存在
            ).count()
            print(f'  FK {field}: 断链={broken}')

    def validate_order_amounts(self):
        """订单金额校验：明细金额之和应 <= 订单总额"""
        from orders.models import Order, OrderItem
        from django.db.models import Sum

        issues = []
        for order in Order.objects.all():
            item_total = order.items.aggregate(
                total=Sum('unit_cost_price', default=0)
            )['total']
            if item_total > order.cost_price and order.cost_price > 0:
                issues.append(order.order_no)

        if issues:
            print(f'❌ 金额不一致的订单: {len(issues)} 个')
        else:
            print('✅ 订单金额校验通过')

    def validate_no_orphans(self):
        """孤儿记录检查"""
        # 无订单的安装单
        orphan_installations = InstallationTask.objects.filter(
            order__isnull=True, source='brand_store'
        ).count()
        print(f'无关联订单的安装单: {orphan_installations}')
```

```sql

CREATE TABLE backup_brands_20260406 AS SELECT * FROM brands;

-- ... 其他表
```

### 6.2 回滚命令

```bash
# 回滚阶段 1（基础数据）
python manage.py migrate_data --rollback --phase 1

# 回滚阶段 2（人员+订单）
python manage.py migrate_data --rollback --phase 2

# 完全回滚（恢复所有备份表）
python manage.py migrate_data --rollback --all
```

```python
def _do_rollback(self, phase, dry_run):
    """按阶段回滚"""
    rollback_map = {
        1: ['brands', 'stores', 'decoration_staff',
            'wood_products', 'alloy_products', 'security_products',
            'suppliers', 'hardware', 'product_categories'],
        2: ['workers', 'foremen', 'wage_standards',
            'orders', 'order_items', 'order_change_logs'],
        3: ['warehouse_products', 'hardware_inventory', 'accessory_inventory'],
        4: ['measurement_tasks', 'installation_tasks', 'maintenance_tasks'],
        5: ['wage_settlements', 'notifications', 'warehouse_transfers'],
    }

    tables = rollback_map.get(phase, [])
    for table in tables:
        backup_table = f'backup_{table}_20260406'
        if dry_run:
            self.stdout.write(f'  [DRY] 将恢复 {table} ← {backup_table}')
        else:
            with connection.cursor() as cursor:
                cursor.execute(f'TRUNCATE TABLE `{table}`')
                cursor.execute(
                    f'INSERT INTO `{table}` SELECT * FROM `{backup_table}`')
            self.stdout.write(f'  ✅ 已恢复 {table}')
```

## 7. 分批迁移策略

### 7.1 迁移顺序

```
阶段 1: 基础数据（无 FK 依赖）
  ├── Brand（品牌）
  ├── Store（门店）
  ├── ProductCategory（产品分类）
  ├── Supplier（供货厂家）
  └── Branch（分公司）

阶段 2: 人员 + 产品（依赖阶段 1）
  ├── DecorationStaff（装企人员）
  ├── Worker（安装师傅）
  ├── Foreman（工头）
  ├── WoodProduct / AlloyProduct / SecurityProduct
  └── Hardware（五金配件）

阶段 3: 订单（依赖阶段 1 + 2）
  ├── Order
  └── OrderItem

阶段 4: 流程数据（依赖阶段 2 + 3）
  ├── MeasurementTask
  ├── InstallationTask
  └── MaintenanceTask

阶段 5: 交易与辅助数据（依赖前面所有阶段）
  ├── WarehouseProduct
  ├── HardwareInventory
  ├── WageSettlement
  └── Notification
```

### 7.2 分批处理

- **默认批大小**: 100 条/批
- **大批量表**: 订单明细建议 500 条/批
- **每批一个事务**: 失败自动回滚当前批
- **错误累积**: 不中断整体迁移，记录到 `migration_errors` 表

```python
from django.db import models


class MigrationLog(models.Model):
    """迁移日志"""
    phase = models.IntegerField('阶段')
    table_name = models.CharField('表名', max_length=100)
    total_count = models.IntegerField('总数')
    migrated_count = models.IntegerField('已迁移数')
    error_count = models.IntegerField('错误数')
    status = models.CharField('状态', max_length=20,
                             choices=[('pending', '待执行'),
                                     ('running', '执行中'),
                                     ('completed', '已完成'),
                                     ('failed', '失败')])
    started_at = models.DateTimeField('开始时间')
    finished_at = models.DateTimeField('完成时间', null=True)
    notes = models.TextField('备注', blank=True)

    class Meta:
        db_table = 'migration_logs'


class MigrationError(models.Model):
    """迁移错误记录"""
    log = models.ForeignKey(MigrationLog, on_delete=models.CASCADE)
    table_name = models.CharField('表名', max_length=100)

    field_name = models.CharField('字段名', max_length=100, blank=True)
    error_type = models.CharField('错误类型', max_length=50)
    error_message = models.TextField('错误信息')
    original_data = models.JSONField('原始数据')
    resolved = models.BooleanField('是否已处理', default=False)

    class Meta:
        db_table = 'migration_errors'
```
